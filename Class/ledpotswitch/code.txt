int switchState=2;
void setup()
{
  pinMode(13, OUTPUT);
  pinMode(switchState, INPUT);
  pinMode(A5, INPUT);
  Serial.begin(9600);
  
}

void loop()
{
  int pot=analogRead(A5);
  Serial.println(pot);
  int switch1 = digitalRead(switchState);
    if (switch1==HIGH) {
     digitalWrite(13, HIGH);
     delay(1000);
     digitalWrite(13,LOW);
     delay(1000);
  }
}
