#include <Servo.h>
// C++ code

Servo s;
int pot = A0;
int servo = 9;

void setup() {
  s.attach(servo);
  Serial.begin(9600);
}

void loop() {
  int potval = analogRead(pot);
  int angle = map(potval, 0, 1023, 0, 180);
  s.write(angle);

  Serial.print("Potentiometer: ");
  Serial.print(potval);
  Serial.print("/n");
  Serial.print("Servo Angle: ");
  Serial.println(angle);

  delay(40);
}
